package lang.object.poly;

import java.util.Objects;
public class Dog {
    int id;
    public  void sound(){
        System.out.println("멍멍");
    }
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Dog dog = (Dog) o;
        return id == dog.id;
    }
}
